void quick_sort(int * tab, int lewy, int prawy)
{
 int i,j,v,buffor;
 i=lewy-1;
 j=prawy+1;
 v = tab[(lewy+prawy)/2];

 if (prawy<=lewy) return;

 while(true)
 {
   while (v>tab[++i]);
   while (v<tab[--j]);

   if (i<=j)
   {
    buffor=tab[i];
    tab[i]=tab[j];
    tab[j]=buffor;
   }
 else break;
 }

 if (i<prawy) quick_sort(tab,i,prawy);
 if (j>lewy) quick_sort(tab,lewy,j);
}

int main()
{
	int *tab, n;

  	cout<<"Ile liczb bedziesz chcia3 posortowaa? ";
  	cin>>n;
  	tab = new int [n]; //przydzielenie pamieci na elementy tablicy
  	//wczytanie liczb
  	for(int i=0;i<n;i++)
    	cin>>tab[i];

  	quick_sort(tab,0, n-1);

  	//wypisanie posortowanych elementów
  	for(int i=0;i<n;i++)
          cout<<tab[i]<<" ";

}
